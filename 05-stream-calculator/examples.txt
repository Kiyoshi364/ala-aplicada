# Always one
# 1, 1, 1, 1, ...
# 1 / (1 - x)
1 x - % ;

# Powers of 2
# 1, 2, 4, 8, ...
# 1 / (1 - 2 * x)
1 2 x * - % ;

# One zero one zero
# 1, 0, 1, 0, ...
# 1 / (1 - x^2)
1 1^2 - % ;

# Zip of reals a and b
# a, b, a, b, ...
# a = 1, 0, 0, ...
1 !1 ;
# b = 2, 0, 0, ...
2 !2 ;
# (a + b * x) / (1 - x^2)
$1 $2 ^1 + 1 1^2 - % * ;

# Fibonacci
# 1 / (1 - x - x^2)
# 1 / (1 - x * (1 + x))
1 1 x + ^1 - % x * ;

# Nats
# 0, 1, 2, 3, ...
# 1 / (1 - x)^2
1 x - % @ * ^1 ;
